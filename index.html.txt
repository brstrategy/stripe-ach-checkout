    // ********************************************

    const checkoutButton = document.getElementById('checkout-button');
    const amountInput = document.getElementById('amount-input');

    checkoutButton.addEventListener('click', async function(e) {
      const amount = parseFloat(amountInput.value) * 100; // Convert to cents (Stripe required)
      if (isNaN(amount) || amount <= 0) {
        alert('Please enter a valid amount.');
        return;
      }

      try {
        // *** 2. YOUR WORKER/SERVERLESS ENDPOINT HERE ***
        // This URL must match the route defined in your Worker config file!
        const response = await fetch('/create-checkout-session', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({ amount: Math.round(amount) }), 
        });

        if (!response.ok) {
          throw new Error('Failed to create Checkout Session.');
        }

        const session = await response.json();

        // 3. Redirect the customer to the Stripe Checkout page
        const result = await stripe.redirectToCheckout({
          sessionId: session.id
        });

        if (result.error) {
          alert(result.error.message);
        }
      } catch (error) {
        console.error('Error:', error);
        alert('There was an error starting checkout. Please try again.');
      }
    });
  </script>
